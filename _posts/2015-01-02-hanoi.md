汉诺塔递归算法的实现

<!-- more -->

	{% highlight c++ %}    
	#include <iostream>
	using namespace std;
 
	//汉诺塔递归算法，将n个盘子从a经过b，移到c,盘子从上往下编号
	void hanoi(int n,char a,char b,char c)
	{
    	if( n==1 )
    	{
        	cout<<"将第"<<n<<"个盘子从"<<a<<"移到"<<c<<endl;
    	}
    	else
    	{
        	hanoi(n-1,a,c,b);
        	cout<<"将第"<<n<<"个盘子从"<<a<<"移到"<<c<<endl;
        	hanoi(n-1,b,a,c);
    	}
	}
 
	int _tmain(int argc, _TCHAR* argv[])
	{
    	int n;
    	cout<<"请输入盘子数量：";
    	while(cin>>n && n>0)           //输入盘子的数量
    	{
        	hanoi(n,'A','B','C');
    	}
    	return 0;
	}
